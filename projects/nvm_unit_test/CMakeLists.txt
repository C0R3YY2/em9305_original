PROJECT(nvm_unit_test C)

    ${NVM_UNITTEST_COMMON_INCLUDES}
)

ARC_LINK_LIBRARIES(${PROJECT_NAME}_spi ${${PROJECT_NAME}_LIBS})
ARC_INCLUDE_DIRECTORIES(${PROJECT_NAME}_spi ${${PROJECT_NAME}_INCLUDES})
